#coverting thai language to english language
d=comments['commentText'].dropna()
eng=[]

for r in d:
    if r:
        f=str(r)

    try:
        g=thai_to_english(f)
        eng.append(g)


    except ValueError:
        continue

from nltk.tokenize import sent_tokenize, word_tokenize
from nltk.stem import PorterStemmer
for g in eng:
    text=text+' '+g


ps = PorterStemmer()
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize

stop_words = set(stopwords.words('english'))

word_tokens = word_tokenize(text)

filtered_sentence = [ps.stem(w) for w in word_tokens if not w in stop_words] # removing useless words

filtered_sentence = []

for w in word_tokens:
    if w not in stop_words:
        filtered_sentence.append(w)
# print filtered_sentence

#classifying comment as positive neutral or negative
from nltk.sentiment.vader import SentimentIntensityAnalyzer
sid = SentimentIntensityAnalyzer()
file_d=[]
for sentence in eng:
    line={}
    line['text']=sentence
    line['sentiment']=''
    ss = sid.polarity_scores(sentence)
    for k in sorted(ss):
        if k=='pos':
            line['positivity']=ss[k]
            if ss[k]>float(0.5):
                line['sentiment']='positive'

        if k=='neg':
            line['negativity']=ss[k]
            if ss[k]>float(0.5):
                line['sentiment']='negative'
        if k=='neu':
            line['neutrality']=ss[k]
            if ss[k]>float(0.5):
                line['sentiment']='neutral'
        if line['sentiment']=='':
            line['sentiment']='neutral'
    file_d.append(line)#saving in dictionary

#creating Word_file dictionary containg all important words
word_file=[]
for word in filtered_sentence:
    line={}
    line['word']=word
    word_file.append(line)

import csv
def WriteDictToCSV(csv_file,csv_columns,dict_data):
    with open(csv_file, 'w') as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=csv_columns)
        writer.writeheader()
        for data in dict_data:
            try:
                writer.writerow(data)
            except UnicodeEncodeError:
                continue

    return
